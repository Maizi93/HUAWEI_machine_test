// 引入 readline 模块 var readline = require('readline'); // 创建 readline 接口实例 var ri = readline.createInterface({ input: process.stdin, output: process.stdout }); ri.on('line',function(line){ var count = 0; var arr = line.split(''); var newArr = []; // 去重 arr.forEach(function(item){ if(newArr.indexOf(item) == -1){ newArr.push(item); } }) // 计数 newArr.forEach(function(item){ if(item.charCodeAt() >= 0 && item.charCodeAt() <= 127){ count ++; } }) console.log(count); })
